\subsection{GGH Verschl{\"u}ssung$^1$}
\begin{frame}
	\frametitle{GGH Verschl{\"u}ssung}
	Private key:
	$R = \begin{pmatrix}
		\mathbf{r}_1 \\ \mathbf{r}_2 
	\end{pmatrix}$
	\\
	Public key:
	$B = \begin{pmatrix}
		\mathbf{b}_1 \\ \mathbf{b}_2 
	\end{pmatrix}$
	\\
	Encrypt \textbf{m}:\\
	\hspace{1.0em} $\mathbf{v} = \mathbf{m}B$\\
	\hspace{1.0em} $\mathbf{c} = \mathbf{v} + \mathbf{e}$\\
	
	Decrypt \textbf{c}:\\
	\hspace{1.0em} $\mathbf{v^{\prime}} = \lfloor\mathbf{c}R^{-1}\rceil  R$\\
	\hspace{1.0em} $\mathbf{m^{\prime}} = \mathbf{v^{\prime}}B^{-1}$
	
	\footnotetext[1]{Basieren auf \cite{Thijs2015_LatticeCryptography}}
\end{frame}
